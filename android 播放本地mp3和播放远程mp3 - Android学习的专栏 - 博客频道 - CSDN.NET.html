  33.                     }  
  34.                     //当播放完音频资源时，会触发onCompletion事件，可以在该事件中释放音频资源，  
  35.                     //以便其他应用程序可以使用该资源:  
  36.                     mediaPlayer.setOnCompletionListener(new OnCompletionListener(){  
  37.                         @Override  
  38.                         public void onCompletion(MediaPlayer mp) {  
  39.                             mp.release();//释放音频资源  
  40.                             stopButton.setEnabled(false);

  42.                         }

  47.                         //开始播放音频  
  48.                         mediaPlayer.start();  
  49.                         playButton.setText("暂停");  
  50.                     } catch (IllegalStateException e) {

  53.                         e.printStackTrace();

  58.                         playButton.setText("播放");  
  59.                     }  
  60.                 }

  62.             }  
  63.         });  
  64.           
  65.         //停止

  75.                 }  
  76.             }  
  77.         });

  79.     }

  94.             return null;  
  95.         }  
  96.         return mp;  
  97.     }

  102.     public MediaPlayer createLocalMp3(){  
  103.         /**

  106.          * 2:播放sdcard卡的文件：mediaPlayer=new MediaPlayer(); 
  107.          *   mediaPlayer.setDataSource("/sdcard/beatit.mp3");//前提是sdcard卡要先导入音频文件

  109.         MediaPlayer mp=MediaPlayer.create(this,R.raw.beatit);  
  110.         mp.stop();  
  111.         return mp;  
  112.     }
